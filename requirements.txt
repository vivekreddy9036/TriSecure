# Installation Instructions:
# ==========================
# 1. Update system:
#    sudo apt-get update && sudo apt-get upgrade
#
# 2. Install system dependencies:
#    sudo apt-get install python3.10 python3.10-dev python3.10-venv
#    sudo apt-get install i2c-tools
#
# 3. Create virtual environment:
#    python3.10 -m venv venv
#    source venv/bin/activate
#
# 4. Install Python dependencies:
#    pip install --upgrade pip wheel setuptools
#    pip install -r requirements.txt
#
# 5. Enable I2C (for NFC):
#    sudo raspi-config  # Then Interface > I2C > Enable
#
# 6. Test hardware:
#    i2cdetect -y 1  # Should show PN532 at address 0x24
#
# 7. Run system:
#    python trisecure/main.py
#
# 8. Deploy as systemd service:
#    sudo cp trisecure.service /etc/systemd/system/
#    sudo systemctl daemon-reload
#    sudo systemctl enable trisecure
#    sudo systemctl start trisecure

# Production Deployment:
# =====================
# For production, use a locked requirements.txt:
#   pip freeze > requirements-lock.txt
# This ensures exact versions across all deployments
